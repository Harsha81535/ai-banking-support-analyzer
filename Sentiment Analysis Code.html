from transformers import pipeline
from tqdm import tqdm
import pandas as pd
import os

# -----------------------------
# 1️⃣  Load sentiment model
# -----------------------------
sentiment_model = pipeline(
    "sentiment-analysis",
    model="cardiffnlp/twitter-roberta-base-sentiment",
    device=-1   # CPU mode
)

# -----------------------------
# 2️⃣  Prepare data
# -----------------------------
texts = df_bank['text'].tolist()
batch_size = 32
checkpoint_every = 1000
checkpoint_file = "banking77_sentiment_checkpoint.csv"

# If resuming from a previous run:
if os.path.exists(checkpoint_file):
    df_checkpoint = pd.read_csv(checkpoint_file)
    done = len(df_checkpoint)
    print(f"🟡 Resuming from checkpoint ({done} rows processed).")
    results = df_checkpoint['sentiment'].tolist()
else:
    done = 0
    results = []
    print("🟢 Starting fresh sentiment labeling.")

# -----------------------------
# 3️⃣  Run batched inference
# -----------------------------
for i in tqdm(range(done, len(texts), batch_size)):
    batch = texts[i:i + batch_size]
    preds = sentiment_model(batch)
    results.extend([p['label'] for p in preds])

    # Periodic checkpoint save
    if (i + batch_size) % checkpoint_every < batch_size:
        partial_df = pd.DataFrame({
            'text': texts[:len(results)],
            'sentiment': results
        })
        partial_df.to_csv(checkpoint_file, index=False)
        print(f"💾 Saved checkpoint at {len(results)} rows.")

# -----------------------------
# 4️⃣  Finalize and Save
# -----------------------------
df_bank['sentiment'] = results
df_bank = df_bank.rename(columns={'label_text': 'intent'})[['text', 'intent', 'sentiment']]
df_bank.to_csv("banking77_with_sentiment.csv", index=False)
print("✅ Done! Saved full BANKING77 with sentiment labels.")
